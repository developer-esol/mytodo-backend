# Git Filter Repo replacements for sensitive data
# This file contains regex patterns to replace secrets with safe placeholders

# AWS Access Key ID pattern
regex:AKIA[0-9A-Z]{16}==>[REDACTED_AWS_ACCESS_KEY_ID]

# AWS Secret Access Key pattern (40 chars base64-like)
regex:[A-Za-z0-9/+=]{40}==>[REDACTED_AWS_SECRET_ACCESS_KEY]

# Twilio Account SID pattern
regex:AC[0-9a-fA-F]{32}==>[REDACTED_TWILIO_SID]

# Twilio Auth Token pattern (32 hex chars)
regex:[0-9a-fA-F]{32}==>[REDACTED_TWILIO_AUTH_TOKEN]

# Google Service Account Private Key pattern
regex:-----BEGIN PRIVATE KEY-----.*-----END PRIVATE KEY-----==>[REDACTED_GOOGLE_SERVICE_ACCOUNT_PRIVATE_KEY]

# Google Service Account Client Email
regex:firebase-adminsdk-[a-z0-9]+@[a-z0-9-]+\.iam\.gserviceaccount\.com==>[REDACTED_SERVICE_ACCOUNT_EMAIL]

# Google Service Account Client ID
regex:"client_id": "[0-9]+"==>"client_id": "[REDACTED_CLIENT_ID]"
